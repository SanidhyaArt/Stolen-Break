  <script>

    const links = {

      US: "https://amzn.to/3PG9EdN",

      UK: "https://www.amazon.co.uk/s?k=KitKat+F1",

      INTL: "https://www.amazon.com/s?k=Formula+1+merchandise"

    function updateLinks() {

      const region = document.getElementById('regionSelect').value;

      document.getElementById('amazonBtn').href = links[region];

    }


    window.onload = () => {

      const tz = Intl.DateTimeFormat().resolvedOptions().timeZone;

      if (tz.includes("America")) document.getElementById('regionSelect').value = "US";

      else if (tz.includes("Europe") || tz.includes("London")) document.getElementById('regionSelect').value = "UK";

      else document.getElementById('regionSelect').value = "INTL";

      updateLinks();

    };


    function checkCode() {

      const val = document.getElementById('batchInput').value.trim();

      if (!val) return;

      

      const prefix = parseInt(val.substring(0, 4));

      const res = document.getElementById('resultBox');

      const stat = document.getElementById('resultStatus');

      const det = document.getElementById('resultDetail');

      

      res.style.display = 'block';

      document.getElementById('ctaWrap').classList.remove('hidden');


      if (prefix >= 6085 && prefix <= 6095) {

        window.va('event', { name: 'BatchChecked', result: 'stolen', prefix: String(prefix) });

        res.className = 'danger p-4 border-l-4';

        stat.innerText = "🚨 Stolen Cargo";

        stat.className = "text-red-600 font-black italic text-2xl uppercase";

        det.innerText = "This batch falls within the reported stolen range. Do not consume.";

      } else {

        window.va('event', { name: 'BatchChecked', result: 'safe', prefix: String(prefix) });

        res.className = 'safe p-4 border-l-4';

        stat.innerText = "✅ Verified Safe";

        stat.className = "text-green-500 font-black italic text-2xl uppercase";

        det.innerText = "This batch is verified as part of the legitimate global supply chain.";

      }

    }
